Python Fallback (any OS, any browser)

If Web Serial isn't available, run this script on your computer — it bridges your Arduino to FCHAN via HTTP.

#!/usr/bin/env python3 # FCHAN Arduino USB Bridge — pip install pyserial requests import serial, requests, time from datetime import datetime, timezone API_URL = "http://localhost:3000/api" API_KEY = "YOUR_API_KEY" PORT = "/dev/ttyUSB0" # Windows: "COM3" BAUD = 9600 SENSOR_MAP = { 'temp':'temperature','temperature':'temperature','tmp':'temperature', 'hum':'humidity','humidity':'humidity', 'soil':'soil_moisture','moisture':'soil_moisture','water sensor value':'soil_moisture', 'ph':'soil_ph','soil_ph':'soil_ph', 'light':'light','lux':'light','intencity':'light','intensity':'light','ldr':'light', 'co2':'co2','wind':'wind','rain':'rainfall','rainfall':'rainfall', } def send(value, stype): r = requests.post(f"{API_URL}/sensors/readings/ingest", json={"api_key": API_KEY, "value": float(value), "recorded_at": datetime.now(timezone.utc).isoformat()}, timeout=5) print("✓" if r.json().get("success") else "✗", stype, value) with serial.Serial(PORT, BAUD, timeout=2) as s: print("Connected. Listening…") while True: line = s.readline().decode(errors='ignore').strip() if ':' not in line: continue label, val = line.rsplit(':', 1) stype = SENSOR_MAP.get(label.strip().lower()) if stype: try: send(float(val.strip()), stype) except: pass

WiFi / HTTP (ESP32 / Any Network Device)

Your ESP32 sends readings directly to the FCHAN API over the network — no bridge needed. Copy the sketch below, paste your WiFi credentials and API key, then upload.

Not tested
Arduino / ESP32 Sketch
// FCHAN WiFi Sensor — ESP32 / ESP8266 // Install: WiFi.h (built-in), HTTPClient.h (built-in), ArduinoJson #include <WiFi.h> #include <HTTPClient.h> #include <ArduinoJson.h> #include <DHT.h> // for DHT11/22 const char* ssid = "YOUR_WIFI_SSID"; const char* password = "YOUR_WIFI_PASSWORD"; const char* apiUrl = http://YOUR_SERVER:3000/api/sensors/readings/ingest; const char* apiKey = "YOUR_API_KEY"; #define DHT_PIN 4 #define DHT_TYPE DHT22 DHT dht(DHT_PIN, DHT_TYPE); void sendReading(float value) { if (WiFi.status() != WL_CONNECTED) return; HTTPClient http; http.begin(apiUrl); http.addHeader("Content-Type", "application/json"); String body = "{\"api_key\":\"" + String(apiKey) + "\",\"value\":" + String(value) + "}"; int code = http.POST(body); http.end(); } void setup() { Serial.begin(9600); dht.begin(); WiFi.begin(ssid, password); while (WiFi.status() != WL_CONNECTED) { delay(500); Serial.print("."); } Serial.println("\nConnected: " + WiFi.localIP().toString()); } void loop() { float temp = dht.readTemperature(); if (!isnan(temp)) sendReading(temp); delay(30000); // send every 30 seconds }
